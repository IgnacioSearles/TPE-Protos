LOGIN_USER_SUCCESS_WRITE -> LOGIN_PASS_READ
LOGIN_USER_INVALID_WRITE -> LOGIN_USER_READ
LOGIN_USER_ERROR_WRITE -> LOGIN_USER_READ

LOGIN_PASS_READ -> LOGIN_PASS_SUCCESS_WRITE | LOGIN_PASS_INVALID_WRITE | LOGIN_PASS_ERROR_WRITE

LOGIN_PASS_ERROR_WRITE -> LOGIN_PASS_READ

MAIN_READ -> EXIT_WRITE | MAIN_ERROR_WRITE
